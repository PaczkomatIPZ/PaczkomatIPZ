//// -- LEVEL 1
//// -- Tables and References

// Creating tables
Table users {
  id int [pk, increment] // auto-increment
  name varchar //nazwa
  box_locker_number int //unikalny numer każdego paczkomatu
}

Table orders {
  id int [pk] //id paczki
  pack_name varchar //nazwa paczki
  sender varchar //nadawca
  receiver varchar //odbiorca
  weight_h_or_low bool //określenie gabarytów paczki (mała,duża)
  open_code int //kod otwarcia
  posting_status bool //status czy paczka jest nadana
  receiving_status bool //status czy paczka jest odebrana
  created_at timestamp //godzina utworzenia paczki/zamówienia
}

Table box_locker {//ogólny zbiór paczkomatów na podstawie numeru paczkomatu
  id int [pk] //numer id paczkomatu
}

Table hiding_place_low { //skrytka na małe gabaryty
  id int [pk]
  box_low_status bool //status czy w skrytce jest paczka
  place int
}

Table hiding_place_high { //skrytka na duże gabaryty
  id int [pk]
  box_high_status bool //status czy w skrytce jest paczka
  place int
}

ref: users.name < orders.id // 1 user moze miec wiele id zamówienia, może zamawiać wielokrotnie
ref: users.box_locker_number - box_locker.id // 1 kod pocztowy ma 1 paczkomat (uproszczenie, bo mogą być np. 3 paczkomaty na 1 kod-pocztowy)
ref: box_locker.id - hiding_place_high.place // 1 paczkomat ma 1 miejsce na duży gabaryt
ref: box_locker.id - hiding_place_low.place // 1 paczkomat ma 1 miejsce na mały gabaryt
ref: users.id - orders.sender // 1 id to 1 nadawca
ref: users.id - orders.receiver // 1 id to 1 odbiorca
