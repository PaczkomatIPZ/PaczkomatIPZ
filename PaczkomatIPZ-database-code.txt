//// -- LEVEL 1
//// -- Tables and References

// Creating tables
Table users {
  id int [pk, increment] // auto-increment
  name varchar
  city int
  country_code int
}

Table orders {
  id int [pk] //id paczki
  name varchar //nazwa paczki
  weight_h_or_low bool //określenie gabarytów paczki (mała,duża)
  open_code int //kod otwarcia
  posting_status bool //status czy paczka jest nadana
  receiving_status bool //status czy paczka jest odebrana
  created_at timestamp //godzina utworzenia paczki/zamówienia
}

Table box_locker {//ogólny zbiór paczkomatów na podstawie kodu miasta
  id int [pk]
  
}

Table hiding_place_low { //skrytka na małe gabaryty
  id int [pk]
  box_low_status bool //status czy w skrytce jest paczka
}

Table hiding_place_high { //skrytka na duże gabaryty
  id int [pk]
  box_high_status bool //status czy w skrytce jest paczka
}

ref: users.name < orders.id // 1 user moze miec wiele id zamówienia, może zamawiać wielokrotnie
ref: users.country_code - box_locker.id // 1 kod pocztowy ma 1 paczkomat (uproszczenie, bo mogą być np. 3 paczkomaty na 1 kod-pocztowy)
ref: box_locker.id - hiding_place_high.id // 1 paczkomat ma 1 miejsce na duży gabaryt
ref: box_locker.id - hiding_place_low.id // 1 paczkomat ma 1 miejsce na mały gabaryt
